	Node* current = NULL; //특정위치를 표현하기 위한 포인터


	new_node = DLL_CreateNode(20);

	new_node = DLL_CreateNode(30);

	while (Tail->next_node != NULL)

//1. 노드의 생성
Node* DLL_CreateNode(int NewData)
{
	Node* NewNode = (Node*)malloc(sizeof(Node));
	NewNode->data = NewData;
	NewNode->next_node = NULL;
	NewNode->pre_node = NULL;

	return NewNode;
}




//7.리스트 출력
void ShowList(Node* Head)
{

	Node* Current = Head;

	while (Current != NULL)
	{
		cout << Current->data << " -> ";
		Current = Current->next_node;
	}
	cout << endl << endl;

}
